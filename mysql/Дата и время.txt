SELECT FIRST_NAME, LAST_NAME, HIRE_DATE FROM employees WHERE HIRE_DATE BETWEEN '1987-06-01' AND '1987-07-30'
//6
SELECT FIRST_NAME, LAST_NAME, HIRE_DATE FROM employees WHERE MONTHNAME(HIRE_DATE) = 'June'
//7
SELECT YEAR(HIRE_DATE) AS YEAR FROM employees  GROUP BY YEAR HAVING COUNT(*) > 5

SELECT DEPARTMENT_NAME, e.COUNT FROM departments d RIGHT JOIN (SELECT DEPARTMENT_ID, COUNT(*) AS COUNT FROM employees WHERE YEAR(HIRE_DATE) = YEAR(CURDATE()) GROUP BY DEPARTMENT_ID) AS e ON d.DEPARTMENT_ID = e.DEPARTMENT_ID
